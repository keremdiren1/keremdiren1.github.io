This code is an example of the usage of a static variable and two get methods.



Dog 2 Code

Dog 2 Code


public class Dog2{
    private String breed;
    private static int dogNumber;

    public Dog2(String b){
        breed = b;
        dogNumber += 1;
    }


    public static int getDogNumber(){
        return dogNumber;
    }

    public String getBreed(){
        return breed;
    }
}

Testing Class


public class Dog2Test{
    public static void main(String[] args){
        Dog2 köpek1 = new Dog2("insan");
        Dog2 köpek2 = new Dog2("Çorba");

        System.out.println(köpek1.getBreed());
        System.out.println(köpek2.getBreed());
        System.out.println(Dog2.getDogNumber());
    }
}
