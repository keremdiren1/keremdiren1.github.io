import java.util.*;
public class Basamak{
    public static void main(String[] args){
        Scanner input = new Scanner(System.in);
        System.out.println("Enter a number:");
        int number = input.nextInt();   //We get an input from the user.
        int basamakNum = 0;   //We initialize the variable that will give us the number of digits.
        int toplamBasamak = 0;   //We initialize the variable that will contain the addition of digits of the number.
        while (number != 0){   //A while loop starts that stops when the number is zero. Since the number decreases a digit each time the while loop iterates, the loop iterates as much as the number of digits the number has.
            basamakNum += 1;   //Since the loop works as much as the number of digits, increasing this number by 1 every time the loop iterates presents us with the number of digits.
            System.out.println(basamakNum + ". basamak " + number % 10);   //This line prints the current digit and the number at that digit.
            toplamBasamak += number % 10;   //This line adds the value of the number at the current digit to this variable, which results in the addition of digits after the loop ends.
            number /= 10;   //This line updates the number variable so that the loop updates every time it iterates.
        }
        System.out.println("Basamak Say覺s覺: " + basamakNum +"\nBasamaklar覺n Toplam覺: " + toplamBasamak);   //This line prints the number of digits and the addition of digits of the inputted number.
    }
}
