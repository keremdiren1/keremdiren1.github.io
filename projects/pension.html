Pension

Tools: CS Code and Java | Source Code: GitHub

In this code, an amount is deducted at the start of each year. Then, at the end of each year, a certain percentage of the current balance is added to the balance. The amount and percentage remain constant from year to year. The starting balance, the amount deducted, the percentage added, and the amount of years are all decided by inputs by the user. The user can do this a total of 5 times before the program finishes.

import java.util.*;
public class Pension{
    public static void main(String[] args){
        Scanner input = new Scanner(System.in);
        double balance;
        double deducted;
        double percentage;
        int year;

        for (int i = 0; i < 5; i++){
            year = -1;
            balance = -1.0;
            percentage = -1;
            deducted = -1.0;   //This part resets some values so that the for loop works as intended.
            System.out.println("What is your balance? (double)");
            balance = input.nextDouble();
            if (balance != -1.0){
                System.out.println("What amount is deducted from your balance? (double)");
                deducted = Math.abs(input.nextDouble());
                if (deducted != -1.0){
                    System.out.println("What percentage do you add to your balance? (double)");
                    percentage = input.nextDouble();
                    if (percentage != -1.0){
                        System.out.println("How many years does this continue? (int)");
                        year = input.nextInt();
                    }
                }
            }   //These ifs get input from the user to assign to variables.
            for (int a = 0; a < year; a++){
                balance -= deducted;
                balance += balance / 100 * percentage;
            }   //This part calculates the final balance.
            if (balance >= 0){
                System.out.println("TRUE");
            }
            else{
                System.out.println("FALSE");
            }   //This part prints out TRUE or FALSE depending on the value of balance.
            System.out.println();
        }
    }
}
